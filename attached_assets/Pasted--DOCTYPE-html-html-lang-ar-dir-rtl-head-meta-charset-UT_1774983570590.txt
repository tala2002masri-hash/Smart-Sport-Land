    <script>
        let DB = {
            m: JSON.parse(localStorage.getItem('sl_m_v4')) || [],
            s: JSON.parse(localStorage.getItem('sl_s_v4')) || [],
            o: JSON.parse(localStorage.getItem('sl_o_v4')) || [{id:1, name:'باقة الافتتاح', price:500, sess:12}],
            f: JSON.parse(localStorage.getItem('sl_f_v4')) || [],
            b: JSON.parse(localStorage.getItem('sl_b_v4')) || [],
            suits: JSON.parse(localStorage.getItem('sl_suits_v4')) || [
                {id:1, size:'S', status:'avail'}, {id:2, size:'S', status:'avail'},
                {id:3, size:'M', status:'avail'}, {id:4, size:'M', status:'avail'},
                {id:5, size:'L', status:'avail'}, {id:6, size:'L', status:'avail'},
                {id:7, size:'XL', status:'avail'}, {id:8, size:'XL', status:'avail'},
                {id:9, size:'XXL', status:'avail'}, {id:10, size:'XXL', status:'avail'}
            ]
        };

        let tempImg = "";

        function showSection(id) {
            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.menu-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            if(event) event.currentTarget.classList.add('active');
            refreshAll();
        }

        function refreshAll() {
            const mOptions = '<option value="">-- اختر مشترك --</option>' + DB.m.map(x=>`<option value="${x.id}">${x.name}</option>`).join('');
            ['l-member', 'p-member', 'b-member', 'port-select'].forEach(id => {
                const el = document.getElementById(id);
                if(el) el.innerHTML = mOptions;
            });
            document.getElementById('m-off').innerHTML = DB.o.map(x=>`<option value="${x.price}" data-sess="${x.sess}">${x.name} (${x.price} ₪)</option>`).join('');
            document.getElementById('m-coach').innerHTML = DB.s.map(x=>`<option>${x.name}</option>`).join('') || '<option>لا يوجد مدربين</option>';
            document.getElementById('b-suit').innerHTML = DB.suits.filter(s=>s.status=='avail').map(s=>`<option value="${s.id}">بدلة ${s.id} [${s.size}]</option>`).join('');
            updateDash(); updateSuitsGrid(); updateStaffTable(); updateFinanceUI(); updateOffersList();
        }

        function saveNewMember() {
            const name = document.getElementById('m-name').value;
            if(!name) return alert("يرجى إدخال الاسم");
            const off = document.getElementById('m-off');
            const newM = {
                id: Date.now(), name, addr: document.getElementById('m-addr').value, goal: document.getElementById('m-goal').value,
                chronic: document.getElementById('m-chr').value, bone: document.getElementById('m-bone').value,
                total: parseFloat(off.value), paid: parseFloat(document.getElementById('m-paid').value) || 0,
                rem_sess: parseInt(off.options[off.selectedIndex].getAttribute('data-sess')),
                coach: document.getElementById('m-coach').value, img: tempImg,
                lab: {h:0, w:0, bmi:0, fat:0, mus:0, chest:0, waist:0, hips:0, thighR:0, thighL:0, armR:0, armL:0},
                history: [], diet: "", train: "", date: new Date().toLocaleDateString('ar-EG')
            };
            newM.debt = newM.total - newM.paid;
            DB.m.push(newM);
            if(newM.paid > 0) DB.f.push({date: new Date().toLocaleString(), desc: `اشتراك جديد: ${newM.name}`, amt: newM.paid, type: 'in'});
            saveDB(); alert("تمت الإضافة بنجاح ✅");
            ['m-name', 'm-addr', 'm-chr', 'm-bone', 'm-paid'].forEach(id => document.getElementById(id).value = "");
            document.getElementById('m-preview').style.display = "none";
            document.getElementById('m-placeholder').style.display = "block";
            tempImg = ""; showSection('dash');
        }

        function confirmBooking() {
            const mId = document.getElementById('b-member').value;
            const suitId = document.getElementById('b-suit').value;
            const bDate = document.getElementById('b-date').value;
            if(!mId || !suitId || !bDate) return alert("اكمل بيانات الحجز!");
            const m = DB.m.find(x => x.id == mId);
            if(m.rem_sess <= 0) return alert("عذراً، لا يوجد جلسات متبقية!");
            m.rem_sess--;
            DB.suits.find(s => s.id == suitId).status = 'booked';
            DB.b.push({mName: m.name, time: bDate, suitId: suitId, suitSize: DB.suits.find(s=>s.id == suitId).size});
            saveDB(); refreshAll(); alert("تم تثبيت الموعد بنجاح ✅");
        }

        function toggleSuitStatus(id) {
            const s = DB.suits.find(x => x.id == id);
            s.status = (s.status === 'avail') ? 'broken' : 'avail';
            saveDB(); refreshAll();
        }

        function calcAIStats() {
            const h = parseFloat(document.getElementById('l-h').value)/100, w = parseFloat(document.getElementById('l-w').value);
            if(h > 0 && w > 0) {
                const bmi = (w / (h*h)).toFixed(1);
                document.getElementById('r-bmi').innerText = bmi;
                document.getElementById('r-fat').innerText = (bmi * 1.2).toFixed(1) + "%";
                document.getElementById('r-mus').innerText = (100 - (bmi * 1.2)).toFixed(1) + "%";
            }
        }

        function saveMemberLab() {
            const m = DB.m.find(x => x.id == document.getElementById('l-member').value);
            if(!m) return;
            const entry = {
                date: new Date().toLocaleDateString('ar-EG'), w: document.getElementById('l-w').value, h: document.getElementById('l-h').value,
                bmi: document.getElementById('r-bmi').innerText, fat: document.getElementById('r-fat').innerText,
                waist: document.getElementById('l-waist').value, chest: document.getElementById('l-chest').value,
                hips: document.getElementById('l-hips').value, thighR: document.getElementById('l-thigh-r').value,
                thighL: document.getElementById('l-thigh-l').value, armR: document.getElementById('l-arm-r').value,
                armL: document.getElementById('l-arm-l').value
            };
            m.lab = entry; if(!m.history) m.history = []; m.history.push(entry);
            saveDB(); loadMemberLab(); alert("تم حفظ القياسات ✅");
        }

        function loadMemberLab() {
            const m = DB.m.find(x => x.id == document.getElementById('l-member').value);
            if(!m) return;
            document.getElementById('l-h').value = m.lab.h || "";
            document.getElementById('l-w').value = m.lab.w || "";
            document.getElementById('l-waist').value = m.lab.waist || "";
            document.getElementById('l-chest').value = m.lab.chest || "";
            document.getElementById('l-hips').value = m.lab.hips || "";
            document.getElementById('l-thigh-r').value = m.lab.thighR || "";
            document.getElementById('l-thigh-l').value = m.lab.thighL || "";
            document.getElementById('l-arm-r').value = m.lab.armR || "";
